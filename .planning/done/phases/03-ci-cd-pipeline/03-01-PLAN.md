**IMPORTANT:** The workflow must:
1. Use `concurrency:` with `${{ github.repository }}` and `cancel-in-progress: true` (BG-08)
2. Build and push Docker image with ONLY SHA tag (BG-07) — NO `latest` tag

**Workflow structure:**

```yaml
name: Deploy to EC2 (Blue-Green)

on:
  push:
    branches:
      - main

# BG-08: Concurrency control — cancels in-progress run when new push arrives
concurrency:
  group: ${{ github.repository }}
  cancel-in-progress: true

env:
  DOCKER_IMAGE: mythicc123/multi-container-service

jobs:
  deploy:
    name: Blue-Green Deploy
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      # Docker setup
      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Login to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}

      # BG-07: Build and push with IMMUTABLE SHA tag only
      # Tag: mythicc123/multi-container-service:sha-a1b2c3d4e5f6...
      # NO 'latest' tag — each deploy is tied to a specific commit
      - name: Build and push Docker image
        uses: docker/build-push-action@v6
        with:
          context: .
          push: true
          tags: ${{ env.DOCKER_IMAGE }}:sha-${{ github.sha }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

      # Install deploy scripts on EC2 (one-time setup)
      - name: Install deploy scripts on EC2
        uses: appleboy/ssh-action@v0.40.0
        with:
          host: ${{ secrets.EC2_HOST }}
          username: ubuntu
          key: ${{ secrets.EC2_SSH_KEY }}
          script: |
            set -euo pipefail
            sudo mkdir -p /tmp/deploy
            # Copy Phase 2 scripts to /tmp/deploy/ via base64 encoding to avoid heredoc quoting issues
            # For each script, encode and decode on remote
            for script in get-active-slot.sh deploy.sh health-check.sh switch-nginx.sh run-deploy.sh; do
              if [[ -f "./scripts/$script" ]]; then
                content=$(cat "./scripts/$script" | base64 -w0)
                echo "$content" | base64 -d | sudo tee "/tmp/deploy/$script" > /dev/null
                sudo chmod +x "/tmp/deploy/$script"
                echo "Installed: /tmp/deploy/$script"
              fi
            done
          timeout: 60s

      # BG-06 + BG-08: Full deploy pipeline with EC2 lock
      - name: Run blue-green deployment
        uses: appleboy/ssh-action@v0.40.0
        with:
          host: ${{ secrets.EC2_HOST }}
          username: ubuntu
          key: ${{ secrets.EC2_SSH_KEY }}
          script: |
            set -euo pipefail

            LOCKFILE="/tmp/blue-green-deploy.lock"
            LOCK_TIMEOUT=300
            LOCK_TTL=600

            # BG-08: Export lock context for traceability
            export GITHUB_RUN_ID="${GITHUB_RUN_ID:-local}"
            export GITHUB_RUN_URL="https://github.com/${{ github.repository }}/actions/runs/${{ github.run_id }}"

            echo "=== Starting deployment (run $GITHUB_RUN_ID) ==="

            # BG-08: Clean stale locks (from crashed/cancelled runs)
            now=$(date -u +%Y-%m-%dT%H:%M:%SZ)
            ttl_at=$(grep '^TTL_AT=' "$LOCKFILE" 2>/dev/null | cut -d= -f2 || echo "")
            if [[ -n "$ttl_at" && "$ttl_at" < "$now" ]]; then
              echo "Cleaning stale lock (TTL expired: ${ttl_at})"
              rm -f "$LOCKFILE"
            fi

            # BG-08: Acquire EC2 lock with flock + TTL
            echo "Acquiring deployment lock..."
            (
              flock -w "$LOCK_TIMEOUT" 9 || { echo "ERROR: Could not acquire lock within ${LOCK_TIMEOUT}s"; exit 1; }
              now=$(date -u +%Y-%m-%dT%H:%M:%SZ)
              ttl_at=$(date -u -d "+${LOCK_TTL} seconds" +%Y-%m-%dT%H:%M:%SZ 2>/dev/null || date -u +%Y-%m-%dT%H:%M:%SZ)
              cat > "$LOCKFILE" << LOCKEOF
LOCK_HELD=1
PID=$$
HOSTNAME=$(hostname)
ACQUIRED_AT=${now}
GITHUB_RUN_ID=${GITHUB_RUN_ID}
GITHUB_RUN_URL=${GITHUB_RUN_URL}
TTL_AT=${ttl_at}
LOCKEOF
              echo "LOCK ACQUIRED: PID=$$ TTL_AT=${ttl_at}"
            ) 9>"$LOCKFILE"

            # BG-06: Run the full deploy pipeline using Phase 2 scripts
            # IMAGE_TAG must match the SHA tag pushed above
            IMAGE_TAG=sha-${{ github.sha }} bash /tmp/deploy/run-deploy.sh
            DEPLOY_STATUS=$?

            # BG-08: Release lock — only if deploy succeeded.
            # On deploy failure, lock is NOT released — TTL will clean it.
            if [[ $DEPLOY_STATUS -eq 0 ]]; then
              rm -f "$LOCKFILE"
              echo "LOCK RELEASED"
            else
              echo "WARNING: Deploy failed (exit $DEPLOY_STATUS) — lock NOT released (TTL will clean it)"
            fi

            exit $DEPLOY_STATUS
          envs: IMAGE_TAG
          timeout: 15m

      # BG-09: Basic smoke test — verify API responds after Nginx switch
      - name: Smoke test API endpoints
        run: |
          sleep 5
          HOST_IP="${{ secrets.EC2_HOST }}"
          echo "Smoke testing API at http://${HOST_IP}"

          # GET /health
          HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" http://${HOST_IP}/health)
          if [[ "$HTTP_CODE" != "200" ]]; then echo "FAIL: GET /health returned $HTTP_CODE"; exit 1; fi
          echo "PASS: GET /health"

          # GET /todos
          HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" http://${HOST_IP}/todos)
          if [[ "$HTTP_CODE" != "200" ]]; then echo "FAIL: GET /todos returned $HTTP_CODE"; exit 1; fi
          echo "PASS: GET /todos"

          # POST /todos — create a todo
          RESPONSE=$(curl -s -X POST http://${HOST_IP}/todos \
            -H "Content-Type: application/json" \
            -d '{"text":"ci-smoke-$(date +%s)","done":false}')
          echo "POST /todos: $RESPONSE"

          # Extract todo id
          TODO_ID=$(echo "$RESPONSE" | grep -o '"_id":"[^"]*"' | head -1 | cut -d'"' -f4)
          if [[ -z "$TODO_ID" ]]; then
            echo "FAIL: Could not extract todo ID from POST response"
            exit 1
          fi

          # PUT /todos/:id — update
          PUT_CODE=$(curl -s -o /dev/null -w "%{http_code}" -X PUT \
            "http://${HOST_IP}/todos/${TODO_ID}" \
            -H "Content-Type: application/json" \
            -d '{"text":"ci-smoke-updated","done":true}')
          if [[ "$PUT_CODE" != "200" ]]; then echo "FAIL: PUT returned $PUT_CODE"; exit 1; fi
          echo "PASS: PUT /todos/${TODO_ID}"

          # DELETE /todos/:id — cleanup
          DEL_CODE=$(curl -s -o /dev/null -w "%{http_code}" -X DELETE \
            "http://${HOST_IP}/todos/${TODO_ID}")
          if [[ "$DEL_CODE" != "200" ]]; then echo "FAIL: DELETE returned $DEL_CODE"; exit 1; fi
          echo "PASS: DELETE /todos/${TODO_ID}"

          echo ""
          echo "SMOKE TEST: ALL PASSED"
        shell: bash
```

**Usage:**
```bash
./ec2-lock.sh acquire   # Acquire lock (blocks until available or timeout)
./ec2-lock.sh release  # Release lock (only if held by current PID)
./ec2-lock.sh status   # Check lock status (exit 0=free, exit 1=held, exit 2=stale)
./ec2-lock.sh cleanup  # Remove stale locks (TTL expired)
```

**Implementation (from research, verbatim):**

```bash
#!/usr/bin/env bash
# scripts/ec2-lock.sh — EC2-side deployment lock management
# Usage:
#   ./ec2-lock.sh acquire  # Acquire lock (blocks until available or timeout)
#   ./ec2-lock.sh release  # Release lock (only if held by us)
#   ./ec2-lock.sh status   # Check if locked (exit 0=free, exit 1=held)
#   ./ec2-lock.sh cleanup  # Remove stale locks (TTL expired)

set -euo pipefail

LOCKFILE="/tmp/blue-green-deploy.lock"
LOCK_TIMEOUT="${LOCK_TIMEOUT:-300}"   # 5 minutes — max wait for lock
LOCK_TTL="${LOCK_TTL:-600}"          # 10 minutes — stale after this
HOSTNAME=$(hostname)

CMD="${1:-}"
[[ -z "$CMD" ]] && { echo "Usage: $0 acquire|release|status|cleanup" >&2; exit 1; }

acquire() {
    local run_id="${GITHUB_RUN_ID:-local}"
    local run_url="${GITHUB_RUN_URL:--}"
    local ttl_at now
    now=$(date -u +%Y-%m-%dT%H:%M:%SZ)
    ttl_at=$(date -u -d "+${LOCK_TTL} seconds" +%Y-%m-%dT%H:%M:%SZ 2>/dev/null || date -u +%Y-%m-%dT%H:%M:%SZ)

    # Atomic acquire using flock. -w TIMEOUT waits TIMEOUT seconds for the lock.
    (
        flock -w "$LOCK_TIMEOUT" 9 || {
            echo "ERROR: Could not acquire lock within ${LOCK_TIMEOUT}s" >&2
            echo "Lock held by:" >&2
            cat "$LOCKFILE" 2>/dev/null | grep -E '^(PID|GITHUB_RUN_ID|TTL_AT)=' >&2 || echo "  (lock file unreadable)" >&2
            exit 1
        }

        # Check if existing lock is stale before overwriting
        if [[ -f "$LOCKFILE" ]]; then
            local existing_ttl
            existing_ttl=$(grep '^TTL_AT=' "$LOCKFILE" | cut -d= -f2 || echo "")
            if [[ -n "$existing_ttl" && "$existing_ttl" > "$now" ]]; then
                echo "ERROR: Lock is actively held (TTL: ${existing_ttl})" >&2
                cat "$LOCKFILE" >&2
                exit 1
            fi
            # Lock is stale — safe to overwrite
        fi

        cat > "$LOCKFILE" <<LOCKEOF
LOCK_HELD=1
PID=$$
HOSTNAME=${HOSTNAME}
ACQUIRED_AT=${now}
GITHUB_RUN_ID=${run_id}
GITHUB_RUN_URL=${run_url}
TTL_AT=${ttl_at}
LOCKEOF
        echo "LOCK ACQUIRED: PID=$$ TTL_AT=${ttl_at}"
    ) 9>"$LOCKFILE"
}

release() {
    if [[ ! -f "$LOCKFILE" ]]; then
        echo "LOCK NOT HELD: no lock file"
        return 0
    fi

    local lock_pid
    lock_pid=$(grep '^PID=' "$LOCKFILE" | cut -d= -f2 || echo "")

    if [[ "$lock_pid" != "$$" ]]; then
        echo "WARNING: Lock held by PID $lock_pid, not $$ — not releasing" >&2
        return 1
    fi

    rm -f "$LOCKFILE"
    echo "LOCK RELEASED"
}

status() {
    if [[ ! -f "$LOCKFILE" ]]; then
        echo "LOCK STATUS: FREE"
        return 0
    fi

    local ttl_at now
    ttl_at=$(grep '^TTL_AT=' "$LOCKFILE" | cut -d= -f2 || echo "")
    now=$(date -u +%Y-%m-%dT%H:%M:%SZ)

    if [[ -n "$ttl_at" && "$ttl_at" < "$now" ]]; then
        echo "LOCK STATUS: STALE (TTL expired at ${ttl_at})"
        return 2   # Distinct exit code for stale
    fi

    echo "LOCK STATUS: HELD"
    grep -E '^(PID|HOSTNAME|ACQUIRED_AT|GITHUB_RUN_ID|TTL_AT)=' "$LOCKFILE"
    return 1
}

cleanup() {
    if [[ ! -f "$LOCKFILE" ]]; then
        echo "CLEANUP: no lock file"
        return 0
    fi

    local ttl_at now
    ttl_at=$(grep '^TTL_AT=' "$LOCKFILE" | cut -d= -f2 || echo "")
    now=$(date -u +%Y-%m-%dT%H:%M:%SZ)

    if [[ -n "$ttl_at" && "$ttl_at" < "$now" ]]; then
        echo "CLEANUP: removing stale lock (TTL expired at ${ttl_at})"
        rm -f "$LOCKFILE"
    else
        echo "CLEANUP: lock is not stale (TTL: ${ttl_at:-unknown}) — skipping"
    fi
}

case "$CMD" in
    acquire) acquire ;;
    release) release ;;
    status)  status ;;
    cleanup) cleanup ;;
    *) echo "Unknown command: $CMD" >&2; exit 1 ;;
esac
```

**Key design:**
- flock atomic acquisition with file descriptor 9
- TTL written atomically inside flock critical section
- Distinct exit codes: acquire (0=ok, 1=fail), status (0=free, 1=held, 2=stale)
- Lock content includes hostname, PID, timestamp, GitHub run ID/URL for diagnostics
